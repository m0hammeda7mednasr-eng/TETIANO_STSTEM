```javascript
  id: item.id,
  title: item.title,
  variant_title: item.variant_title,
  quantity: item.quantity,
  price: item.price,
  sku: item.sku,
  vendor: item.vendor,
  fulfillment_status: item.fulfillment_status,
  // ... كل التفاصيل
}));

// Extract COMPLETE shipping address
order.shipping_address = {
  first_name: orderData.shipping_address.first_name,
  last_name: orderData.shipping_address.last_name,
  address1: orderData.shipping_address.address1,
  // ... كل التفاصيل
};

// Extract COMPLETE customer info
order.customer_info = {
  id: orderData.customer.id,
  email: orderData.customer.email,
  orders_count: orderData.customer.orders_count,
  total_spent: orderData.customer.total_spent,
  // ... كل التفاصيل
};

// Extract shipping lines, discounts, taxes, refunds, fulfillments
order.shipping_lines = orderData?.shipping_lines || [];
order.discount_codes = orderData?.discount_codes || [];
order.tax_lines = orderData?.tax_lines || [];
order.refunds = orderData?.refunds || [];
order.fulfillments = orderData?.fulfillments || [];

// Extract payment details
order.payment_gateway_names = orderData?.payment_gateway_names || [];
order.processing_method = orderData?.processing_method || null;

// Extract source information
order.source_name = orderData?.source_name || "";
order.referring_site = orderData?.referring_site || null;
order.landing_site = orderData?.landing_site || null;

// ... وكل التفاصيل الأخرى
```

## كيفية الاستخدام

### 1. عرض تفاصيل الطلب:

1. اذهب إلى **"الطلبات"**
2. اضغط على **"عرض"** بجانب أي طلب
3. ستشوف **كل التفاصيل** في صفحة واحدة:
   - معلومات العميل الكاملة
   - عنوان الشحن الكامل
   - عنوان الفواتير
   - تفاصيل المنتجات
   - معلومات الدفع
   - معلومات الشحن
   - الخصومات
   - التصنيفات
   - ملاحظة العميل
   - مصدر الطلب
   - التعليقات

### 2. إضافة تعليق:

1. في صفحة تفاصيل الطلب
2. اكتب التعليق في المربع
3. اضغط **"إرسال"**
4. التعليق سيظهر فوراً
5. سيتم مزامنته مع شوبيفاي تلقائياً

### 3. تغيير حالة الطلب:

1. في صفحة تفاصيل الطلب
2. اختر الحالة الجديدة من القائمة المنسدلة
3. سيتم التحديث فوراً
4. سيتم مزامنته مع شوبيفاي تلقائياً

## مشاكل محتملة وحلولها

### المشكلة: لا تظهر بعض التفاصيل

**الحل:**

1. تأكد من وجود البيانات في شوبيفاي
2. قم بمزامنة الطلبات مرة أخرى
3. تحقق من console للأخطاء

### المشكلة: عنوان الفواتير لا يظهر

**الحل:**

- بعض الطلبات قد لا يكون لها عنوان فواتير منفصل
- في هذه الحالة، عنوان الشحن هو نفسه عنوان الفواتير

### المشكلة: لا توجد معلومات شحن

**الحل:**

- بعض الطلبات قد لا تكون قد شُحنت بعد
- معلومات الشحن تظهر فقط بعد إنشاء الشحنة

### المشكلة: لا توجد أكواد خصم

**الحل:**

- الكارت يظهر فقط إذا كان هناك أكواد خصم مستخدمة
- إذا لم يستخدم العميل كود خصم، الكارت لن يظهر

## ملاحظات مهمة

### ✅ البيانات الكاملة

- كل البيانات تُستخرج من حقل `data` في قاعدة البيانات
- هذا الحقل يحتوي على كل تفاصيل الطلب من شوبيفاي
- لا يتم فقدان أي معلومات

### ✅ العرض الذكي

- الكروت تظهر فقط إذا كانت البيانات موجودة
- إذا لم يكن هناك عنوان فواتير، الكارت لن يظهر
- إذا لم يكن هناك خصومات، الكارت لن يظهر

### ✅ التنظيم

- التفاصيل الرئيسية في المنتصف
- المعلومات الجانبية في الـ Sidebar
- تنظيم منطقي وسهل التصفح

### ✅ الأداء

- البيانات تُحمل مرة واحدة فقط
- لا توجد طلبات إضافية للسيرفر
- سرعة عرض ممتازة
